    ```bash
    curl -sSfL https://install.command.ockam.io | bash \
        && source "$HOME/.ockam/env"
    ```
  </Step>
  <Step title="Deploy your first agent">
    ```bash
    mkdir hello && cd hello && ockam
    ```
  </Step>
  <Step title="Chat with your agent">
    ```
    > Tell me a tale
    ```
  </Step>

<CodeGroup>

```python images/main/main.py
from ockam import Agent, Node, Repl
from sys import argv


async def main(node):
    agent = await Agent.start(node, "You are Jack Sparrow.")
    await Repl.start(agent, argv[1])


Node.start(main)
```


```yaml ockam.yaml
name: hello
pods:
  - name: main-pod
    containers:
      - name: main
        image: main
        args: [localhost:9000]
    portals:
      outlets:
        - to: localhost:9000
```
