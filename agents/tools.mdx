```python
from ockam import Agent, Model, Node, McpTool, McpClient, Repl, Knowledge
from sys import argv

async def main(node):
    agent = await Agent.start(
        node=node,
        name="web_search_assistant",
        instructions="""
            You are a web search assistant.
            
            When you are given a question, decide if it is a
            local or a web search question.

            Depending on the type of question, invoke the appropriate tool
            and explain its response.
        """,
        tools=[
            McpTool("brave_search", "brave_web_search"),
            McpTool("brave_search", "brave_local_search"),
        ],
    )

    await Repl.start(agent, argv[1])

Node.start(
    main,
    mcp_clients=[
        McpClient(name="brave_search", address="http://localhost:8000/sse"),
    ],
)
```
