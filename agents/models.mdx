```python images/main/main.py
from ockam import Agent, Model, Node, Repl
from sys import argv


async def main(node):
    agent = await Agent.start(
        node=node,
        instructions="You are Henry, an expert legal assistant",
        model=Model("claude-3-7-sonnet-v1"),
    )

    await Repl.start(agent, argv[1])


Node.start(main)
```
